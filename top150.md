### 88.合并两个有序数组
``` java
class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int cur = m + n - 1;
        int p1 = m - 1, p2 = n - 1;
        while (cur >= 0) {
            if (p1 < 0) {
                nums1[cur] = nums2[p2--];
            }
            else if (p2 < 0) {
                nums1[cur] = nums1[p1--];
            }
            else if (nums1[p1] > nums2[p2]) {
                nums1[cur] = nums1[p1--];
            } 
            else {
                nums1[cur] = nums2[p2--];
            }
            cur--;
        }
    }
}
```

### 27.移除元素
``` java
class Solution {
    public int removeElement(int[] nums, int val) {
        int slow = 0;
        // 定义快慢指针，快指针便利数组，如果不是需要删除的元素，则利用慢指针覆盖入原数组
        for (int fast = 0; fast < nums.length; fast++) {
            if (nums[fast] != val) {
                nums[slow] = nums[fast];
                ++slow; 
            }
        }
        return slow;
    }
}
```

### 26.删除有序数组中的重复项
``` java
class Solution {
    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        int fast = 1, slow = 1;
        // 定义快慢指针，快指针便利数组，如果与前一个元素不相同，则利用慢指针覆盖入原数组
        while (fast < n) {
            if (nums[fast] != nums[fast - 1]) {
                nums[slow] = nums[fast];
                ++slow;
            }
            ++fast;
        }
        return slow;
    }
}
```

### 380.O(1) 时间插入、删除和获取随机元素
``` go
type RandomizedSet struct {
    // 数组用来记录元素的值
    nums    []int
    // map的key是元素的值，value用来记录元素的下标
    // map[value] = index
    indices map[int]int
}

func Constructor() RandomizedSet {
    return RandomizedSet{[]int{}, map[int]int{}}
}

func (this *RandomizedSet) Insert(val int) bool {
    // map[key] 返回两个值
    // 第一个值：如果key对应的value存在，则返回value，如果不存在则返回零值
    // 第二个值：如果key对应的value存在，则返回true，如果不存在则返回false
    if _, ok := this.indices[val]; ok {
        return false
    }
    // index改为目前元素数量
    this.indices[val] = len(this.nums)
    // 再把val加入到数组中
    this.nums = append(this.nums, val)
    return true
}


func (this *RandomizedSet) Remove(val int) bool {
    // 获取要删除值的index    
    idx, ok := this.indices[val]
    if !ok {
        return false
    }
    // 获取目前数组最后一位的value
    last := this.nums[len(this.nums) - 1]
    // 将原本要删除位置的value改成最后一位的value
    this.nums[idx] = last
    // 因为把最后一位移到前面删除元素的位置了，所以对应map中记录的index也要更改，改成删除元素的index
    this.indices[last] = idx
    // 删除数组中的最后一位，因为最后一位已经往前移(复制)到原本删除元素的位置了，最后一位是重复的
    this.nums = this.nums[:len(this.nums) - 1];
    // 再把map中要删除元素删除
    delete(this.indices, val)
    return true
}


func (this *RandomizedSet) GetRandom() int {
    return this.nums[rand.Intn(len(this.nums))]
}
```

## 双指针
### 167.两数之和 II - 输入有序数组
``` java
class Solution {
    // 左右指针，因为是递增序列， 如果左右加起来大了，则右指针左移，反之左指针右移
    public int[] twoSum(int[] numbers, int target) {
        int n = numbers.length;
        int left = 0, right = n - 1;
        while (left < right) {
            if (numbers[left] + numbers[right] == target) 
                return new int[]{left + 1, right + 1};
            else if (numbers[left] + numbers[right] > target) 
                right--;
            else 
                left++;
        }
        return new int[2];
    }
}
```
